\newcommand\blankpage{%
   \afterpage{
      \null
      \thispagestyle{empty}
      \newpage
   }
   \clearpage
}

\newcommand\cleartorightpage{
   \afterpage{
      \null
      \thispagestyle{empty}
      \ifodd\value{page}
         \clearpage \null
      \fi
      \newpage
   }
   \clearpage
}

\newcommand\cleartoleftpage{
   \afterpage{
      \null
      \thispagestyle{empty}
      \ifodd\value{page}
      \else
         \clearpage \null
      \fi
      \newpage
   }
   \clearpage
}
